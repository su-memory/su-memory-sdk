#### 代码优化示例

```python
# lite.py 优化版本

class SuMemoryLite:
    """轻量级SDK - 优化版"""
    
    def __init__(self, max_memories: int = 10000):
        self.max_memories = max_memories
        self._memories: List[Dict] = []
        self._index: Dict[str, Set[str]] = {}  # 使用Set去重
    
    def _extract_keywords(self, text: str) -> List[str]:
        """优化的关键词提取"""
        # 1. 简单分词
        words = text.split()
        
        # 2. 停用词过滤
        stop_words = {'的', '了', '和', '是', '在', '有', '我', '你', '他'}
        keywords = [w for w in words if len(w) > 1 and w not in stop_words]
        
        return keywords
    
    def query(self, query: str, top_k: int = 5) -> List[Dict]:
        """优化的查询"""
        keywords = self._extract_keywords(query)
        
        # TF-IDF 评分
        scores = {}
        for keyword in keywords:
            if keyword in self._index:
                idf = math.log(len(self._memories) / len(self._index[keyword]))
                for memory_id in self._index[keyword]:
                    scores[memory_id] = scores.get(memory_id, 0) + idf
        
        # 归一化并排序
        if scores:
            max_score = max(scores.values())
            scores = {k: v/max_score for k, v in scores.items()}
        
        return self._get_top_k(scores, top_k)
```
