```python
def find_causal_pairs(
    self, memories, 
    use_statistical: bool = False,  # 默认关闭，向后兼容
    energy_bus=None,
    index: Optional[Dict] = None,
) -> List[Tuple[Dict, Dict, str, float]]:

    pairs = self._keyword_causal_pairs(memories)
    
    # 2. 统计路径 (新增)
    if use_statistical and SPECTRAL_AVAILABLE and len(memories) >= 10:
        dag = GaussianDAG(memories, index, energy_bus)
        stat_edges = dag.discover_hidden_edges()
        for edge in stat_edges:
            i, j = edge["cause_idx"], edge["effect_idx"]
            # 去重: 避免与关键词结果重复
            if not _is_duplicate(pairs, memories[i]["id"], memories[j]["id"]):
                pairs.append((
                    memories[i], memories[j],
                    f"{edge['verdict']}_{edge.get('energy_relation', '')}",
                    edge["confidence"]
                ))
    
    return sorted(pairs, key=lambda x: x[3], reverse=True)
```

```python
def _is_duplicate(pairs, id_a, id_b) -> bool:
    """检查因果对是否已在列表中。"""
    for p in pairs:
        pid_a = p[0].get("id", "")
        pid_b = p[1].get("id", "")
        if (pid_a == id_a and pid_b == id_b) or (pid_a == id_b and pid_b == id_a):
            return True
    return False
```

```python
@dataclass
class GaussianDistribution:
    """
    高斯分布 — 连续因果效应的共轭先验。
    
    平行于已有 BetaDistribution:
    - BetaDistribution → 二项概率 (事件是否发生)
    - GaussianDistribution → 连续效应 (效应有多大)
    
    Normal-Normal 共轭:
    μ_post = (μ₀/σ₀² + n·x̄/σ²) / (1/σ₀² + n/σ²)
    σ²_post = 1 / (1/σ₀² + n/σ²)
    """
    mu: float = 0.0
    sigma: float = 1.0
    n_observations: int = 0
    
    @property
    def mean(self) -> float: ...
    @property
    def variance(self) -> float: ...
    @property
    def precision(self) -> float: ...
    
    def pdf(self, x: float) -> float: ...
    def cdf(self, x: float) -> float: ...
    
    def credible_interval(self, prob: float = 0.95) -> Tuple[float, float]:
        """μ ± z·σ, z 取决于 prob。"""
        ...
    
    def update(self, sample_mean: float, sample_std: float, 
               n: int) -> 'GaussianDistribution':
        """Normal-Normal 共轭更新，返回 self (链式调用)。"""
        ...
    
    def to_dict(self) -> Dict: ...
    
    @classmethod
    def from_dict(cls, d: Dict) -> 'GaussianDistribution': ...
```

```python
class BayesianCausal:
    """
    贝叶斯因果推断 — 能量先验增强版。
    
    H₀: 无因果效应 (ρ = 0)
    H₁: 有因果效应 (ρ ≠ 0)
    
    先验融入能量中心:
    - 生关系 → μ₀ = 0.3, σ₀ = 0.5 (正效应预期)
    - 克关系 → μ₀ = 0.0, σ₀ = 0.3 (保守)
    - 无关系 → μ₀ = 0.0, σ₀ = 1.0 (无信息先验)
    """
    
    def __init__(self, energy_bus=None):
        self._bus = energy_bus
        self._posteriors: Dict[str, GaussianDistribution] = {}
        self._test_history: List[Dict] = []
    
    def causal_hypothesis_test(
        self,
```

```python
class GaussianDAG:
    # 新增属性
    _bayesian: Optional[BayesianCausal] = None
    
    def with_bayesian_quantification(self, bayesian: BayesianCausal):
        """注入 BayesianCausal 以启用后验量化输出。"""
        self._bayesian = bayesian
    
    def discover_hidden_edges(self, ...) -> List[Dict]:
        edges = self._discover_raw_edges(...)
        
        # 贝叶斯量化 (如果启用)
        if self._bayesian is not None:
            for edge in edges:
                eid = f"{edge['cause_idx']}_{edge['effect_idx']}"
                posterior = self._bayesian.causal_hypothesis_test(
                    eid, edge["rho"], self._n_effective,
                    edge.get("energy_relation"),
                )
                edge["posterior_mean"] = posterior["posterior_mean"]
                edge["posterior_std"] = posterior["posterior_std"]
                edge["credible_interval_95"] = posterior["credible_interval_95"]
                edge["bayes_factor"] = posterior["bayes_factor"]
                edge["conclusion"] = posterior["conclusion"]
        
        return edges
```
