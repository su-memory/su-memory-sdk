## 一、M4 — 噪声梯度验证 (v3.5.0-p0) · 细化版

**工期**: 3–4 天  
**目标**: 在检索结果中逐步注入噪声，量化 FourierCausal + GaussianDAG 的因果发现鲁棒性  
**战略意义**: 这是 v3.6.0 参数化训练投入的决策开关——如果 3N 噪声下因果发现退化超过 50%，参数化转型必要性大幅上升

> **背景**: MEMO 论文 (Table 13) 揭示了检索噪声对推理精度的非线性影响——0N→1N 可能降 5–10%，1N→2N 可能降 15%+。v3.5.0 需要量化 su-memory 在这些噪声水平下的表现，作为 v3.6.0 参数化训练前最后的"检索范式极限"测量。

### 细化设计原则

1. **哈希确定性**: 所有噪声生成基于固定种子 (seed=42) + 内容哈希，确保每次运行严格可复现
2. **三级噪声递增**: 0N (纯信号) → 1N (轻度混淆) → 2N (中度噪声) → 3N (重度噪声+对抗)
3. **噪声隔离**: 噪声记忆使用独立 ID 前缀 `n{对序号}_{k}`，不与真实记忆混合统计
4. **零侵入**: 不修改任何 SDK 源码，所有噪声逻辑仅在 benchmark 层

```python
"""
su-memory v3.5.0 — Retrieval Noise Generator
=============================================
按三种策略生成噪声记忆: 语义相似噪声、随机噪声、对抗噪声。

设计原则:
- 哈希确定性: 所有噪声基于 seed + content_hash，严格可复现
- 噪声隔离: 生成噪声不含原始因果信号
- 零依赖: 纯 Python stdlib + numpy
"""

import hashlib
import random
import numpy as np
from typing import List, Tuple

# ==========================================================================
# 语义同义词库 (15 组，覆盖经济/科技/政策/自然/健康领域)
# ==========================================================================

SEMANTIC_SYNONYMS: List[Tuple[str, List[str]]] = [
    # ── 经济金融 ──
    ("物价", ["价格", "费率", "成本"]),
    ("上涨", ["上升", "增长", "攀升", "提高", "走高"]),
    ("下降", ["下滑", "走低", "回落", "降低", "下行"]),
    ("消费", ["支出", "购买", "花销", "购置"]),
    ("公司", ["企业", "机构", "单位", "组织"]),
    ("营收", ["收入", "营业额", "进账", "流水"]),
    ("市场", ["行情", "大盘", "市况", "交易"]),
    ("裁员", ["缩编", "人员优化", "人力调整", "岗位裁撤"]),
    # ── 科技 ──
    ("研发", ["科研", "技术开发", "产品研究", "创新"]),
    ("突破", ["进展", "创新", "飞跃", "跨越"]),
    ("数据", ["指标", "参数", "统计", "测量值"]),
    # ── 政策 ──
    ("政策", ["规定", "制度", "条例", "法规"]),
    ("利率", ["利息率", "贷款成本", "资金价格", "贴现率"]),
    # ── 自然/环境 ──
    ("气温", ["温度", "热度", "气象指标", "环境温度"]),
    ("极端", ["异常", "罕见", "非常规", "超常"]),
]

# ==========================================================================
# 随机中文 token 种子 (固定 50 个，用于构建无意义但语法通顺的句子)
# ==========================================================================

RANDOM_NOUNS = [
    "报告", "系统", "项目", "会议", "方案", "流程", "标准", "资源",
    "部门", "平台", "模块", "接口", "协议", "框架", "策略", "机制",
    "配置", "版本", "需求", "文档", "组件", "引擎", "节点", "通道",
]

RANDOM_VERBS = [
    "完成", "更新", "优化", "调整", "发布", "配置", "部署", "验证",
    "测试", "审核", "提交", "执行", "处理", "生成", "同步", "初始化",
]

RANDOM_ADJECTIVES = [
    "新的", "主要", "重要", "关键", "基本", "核心", "标准", "通用",
    "高级", "基础", "完整", "高效", "稳定", "安全", "可靠", "灵活",
]

# ==========================================================================
# 对抗噪声模板 (6 种变换策略 —— 保留关键词但换新语境)
# ==========================================================================

ADVERSARIAL_STRATEGIES = [
    "{keyword_a}和{keyword_b}的关系研究取得了{progress}",
    "关于{keyword_a}的最新分析显示{trend}趋势",
    "{keyword_b}领域专家认为{opinion}",
    "{keyword_a}的历史数据显示{data_point}",
    "本周{keyword_b}相关话题讨论热度{direction}",
    "{keyword_a}在不同地区的表现差异显著",
]

ADVERSARIAL_FILL = {
    "progress": ["初步成果", "阶段性进展", "预期突破", "新发现"],
    "trend": ["持续向好", "波动加剧", "稳定增长", "缓慢回升"],
    "opinion": ["需要审慎评估", "前景乐观", "存在不确定性", "值得关注"],
    "data_point": ["呈现周期性规律", "显著偏离均值", "符合正态分布", "具有统计意义"],
    "direction": ["明显上升", "略微下降", "基本持平", "大幅波动"],
}


class NoiseGenerator:
    """
    检索噪声生成器 —— 哈希确定性 + 三级策略。

    用法:
        ng = NoiseGenerator(seed=42)
        # 0N: 不生成噪声
        assert ng.generate(["A", "B"], noise_level=0) == []
        # 1N: 每条记忆 1 条语义噪声
        noises = ng.generate(["物价上涨3.5%"], noise_level=1, noise_mode="semantic")
        assert len(noises) == 1
    """

    def __init__(self, seed: int = 42):
        self._master_seed = seed
        self._rng = random.Random(seed)

    # ------------------------------------------------------------------
    # 哈希确定性随机
    # ------------------------------------------------------------------

    def _hash_deterministic(self, content: str, salt: str = "") -> random.Random:
        """
        基于 content + salt 的确定性随机生成器。

        同一 content + salt 永远产生相同的随机序列，
        不同 content 产生不同但可复现的序列。
        """
        h = hashlib.sha256(
            f"{self._master_seed}:{content}:{salt}".encode()
        ).digest()
        seed_int = int.from_bytes(h[:8], "big") % (2 ** 31)
        return random.Random(seed_int)

    # ------------------------------------------------------------------
    # 主入口: 按噪声等级生成
    # ------------------------------------------------------------------

    def generate(
        self,
        ground_truth: List[str],
        noise_level: int,
        noise_mode: str = "semantic",
    ) -> List[str]:
        """
        为 ground_truth 列表注入噪声。

        Args:
            ground_truth: 真实因果记忆列表 (如 [cause, effect])
            noise_level: 0N | 1N | 2N | 3N
            noise_mode: noise_level=3 时自动切换到 "adversarial"
                        其他等级默认 "semantic"

        Returns:
            噪声记忆列表 (不包含原始 ground_truth)

        噪声注入协议:
            0N → []
            1N → 每条 ground_truth 生成 1 条 semantic 噪声
            2N → 每条 ground_truth 生成 2 条 semantic 噪声
            3N → 每条 ground_truth 生成 2 条 semantic + 1 条 adversarial
        """
        if noise_level <= 0:
            return []

        noises: List[str] = []

        for original in ground_truth:
            if noise_level >= 1:
                noises.append(self._generate_semantic_noise(original))
            if noise_level >= 2:
                noises.append(self._generate_semantic_noise(original + "_v2"))
            if noise_level >= 3:
                noises.append(self._generate_adversarial_noise(original))

        return noises

    # ------------------------------------------------------------------
    # 策略 1: 语义噪声 (50-70% 同义词替换)
    # ------------------------------------------------------------------

    def _generate_semantic_noise(self, original: str) -> str:
        """
        对原文进行同义词替换。

        策略:
        - 扫描原文，对匹配到的词做替换
        - 替换率控制在 50-70% (至少保留 30% 原始字符)
        - 替换后的句子语义相似但因果信号被破坏

        示例:
            原文: "物价指数同比上涨百分之三点五"
            噪声: "费率指数同比上升百分之三点五"  ← "物价"→"费率", "上涨"→"上升"
        """
        rng = self._hash_deterministic(original, "semantic")
        result = original

        for keyword, synonyms in SEMANTIC_SYNONYMS:
            if keyword in result:
                replacement = rng.choice(synonyms)
                result = result.replace(keyword, replacement, 1)

        return result

    # ------------------------------------------------------------------
    # 策略 2: 随机噪声 (无因果信号的中文句子)
    # ------------------------------------------------------------------

    def _generate_random_noise(self, original: str = "") -> str:
        """
        生成完全随机的噪声句子。

        格式: [形容词] + [名词] + [动词] + [名词] + 状态描述
        确保不通往任何因果记忆。
        """
        rng = self._hash_deterministic(original or str(self._rng.random()), "random")

        adj = rng.choice(RANDOM_ADJECTIVES)
        noun1 = rng.choice(RANDOM_NOUNS)
        verb = rng.choice(RANDOM_VERBS)
        noun2 = rng.choice(RANDOM_NOUNS)

        return f"{adj}{noun1}{verb}{noun2}第{rng.randint(1,999)}号任务"

    # ------------------------------------------------------------------
    # 策略 3: 对抗噪声 (共享关键词但语境无关)
    # ------------------------------------------------------------------

    def _generate_adversarial_noise(self, original: str) -> str:
        """
        生成对抗噪声 —— 与原文共享关键词但语义无关。

        最危险的噪声类型: 与真记忆共享词汇，
        容易在向量空间中靠近真记忆但因果方向不同。

        策略:
        1. 从原文提取 2 个关键词
        2. 将其填入无关语境模板

        示例:
            原文: "物价指数同比上涨百分之三点五"
            噪声: "物价和消费的关系研究取得了初步成果"  ← 共享"物价"但无因果
        """
        rng = self._hash_deterministic(original, "adversarial")

        # 提取关键词: 优先从 SEMANTIC_SYNONYMS 中找匹配
        keywords = []
        for keyword, _ in SEMANTIC_SYNONYMS:
            if keyword in original and len(keywords) < 2:
                keywords.append(keyword)

        # 如果不足 2 个，随机取字符片段
        while len(keywords) < 2 and len(original) > 4:
            start = rng.randint(0, len(original) - 3)
            fragment = original[start:start + 2]
            if fragment not in keywords:
                keywords.append(fragment)

        if len(keywords) < 1:
            keywords = ["系统", "数据"]

        # 填入对抗模板
        template = rng.choice(ADVERSARIAL_STRATEGIES)
        fill_key = rng.choice(list(ADVERSARIAL_FILL.keys()))
        fill_val = rng.choice(ADVERSARIAL_FILL[fill_key])

        result = template.format(
            keyword_a=keywords[0] if len(keywords) > 0 else "系统",
            keyword_b=keywords[1] if len(keywords) > 1 else "数据",
            **{fill_key: fill_val},
        )

        # 确保与原文不同
        if result == original:
            return self._generate_random_noise(original)

        return result

    # ------------------------------------------------------------------
    # 生成噪声的批量版本 (直接返回可直接插入的记忆字典列表)
    # ------------------------------------------------------------------

    def generate_as_memories(
        self,
        ground_truth_ids: List[str],
        ground_truth_contents: List[str],
        noise_level: int,
    ) -> List[dict]:
        """
        批量生成噪声记忆字典，可直接插入记忆引擎。

        Args:
            ground_truth_ids: 真实记忆的 ID 列表
            ground_truth_contents: 真实记忆的内容列表
            noise_level: 噪声等级

        Returns:
            [{"id": "n_{gt_id}_{k}", "content": noise_text}, ...]
        """
        noise_memories = []
        for gt_id, gt_content in zip(ground_truth_ids, ground_truth_contents):
            noises = self.generate([gt_content], noise_level=noise_level)
            for k, noise_text in enumerate(noises):
                noise_memories.append({
                    "id": f"n_{gt_id}_{k}",
                    "content": noise_text,
                })
        return noise_memories
```

@dataclass
class NoiseGradientResult:
    """v3.5.0 — 噪声梯度测试结果"""
    # ── 各噪声水平得分 ──
    base_score: float = 0.0          # 0N 基线分 (纯信号)
    score_1n: float = 0.0            # 1N: 轻度语义噪声
    score_2n: float = 0.0            # 2N: 中度语义噪声
    score_3n: float = 0.0            # 3N: 重度噪声 (semantic + adversarial)

    # ── 逐级退化率 ──
    degradation_1n: float = 0.0      # 0N → 1N 衰减量
    degradation_2n: float = 0.0      # 0N → 2N 衰减量
    degradation_3n: float = 0.0      # 0N → 3N 衰减量

    # ── 综合抗噪能力 (加权) ──
    noise_robustness_score: float = 0.0
    # 公式: 3N保持率×50% + 1N保持率×30% + 2N保持率×20%

    # ── 分策略细粒度 ──
    semantic_noise_resistance: float = 0.0   # 纯语义噪声抗性 (1N-2N 平均保持率)
    adversarial_noise_resistance: float = 0.0  # 对抗噪声额外退化 (2N→3N 的增量退化)

def bench_causal_inference_noise_gradient() -> NoiseGradientResult:
    """
    v3.5.0 — 噪声梯度因果检测。

    设计原理 (对齐 MEMO Table 13 噪声鲁棒性分析):

    噪声等级定义:
    ┌────────┬─────────────────────────────────────────┬──────────────┐
    │ Level  │ 描述                                    │ 总记忆数     │
    ├────────┼─────────────────────────────────────────┼──────────────┤
    │   0N   │ 仅 10 对因果记忆 (基线)                  │     20       │
    │   1N   │ 每条因果记忆后注入 1 条语义噪声           │     40       │
    │   2N   │ 每条因果记忆后注入 2 条语义噪声           │     60       │
    │   3N   │ 每条因果记忆后注入 2 条语义 + 1 条对抗   │     80       │
    └────────┴─────────────────────────────────────────┴──────────────┘

    噪声注入方式: 交错 (interleaved)
    真实记忆 C₁, E₁, C₂, E₂, ..., C₁₀, E₁₀
    → 噪声插入为 N₁₁, N₁₂, N₂₁, N₂₂, ... (按记忆序号)
    即真实记忆在前，噪声接续在后，保证噪声不插入因果对之间。

    测试目标:
    - 在每个噪声水平下，GaussianDAG 能否正确定向检测 10 对隐藏因果
    - 测量 FourierCausal 的频域周期混淆过滤在噪声下的有效性
    - 测量 BayesianCausal 的后验量化在噪声下是否保持稳健

    阈值设计:
    - CausalEngine(min_confidence=0.3): 低于此阈值的噪声对会被自动过滤
    - 期望: 0N 下 ≥ 8/10, 3N 下 ≥ 4/10 (检索范式极限)
    """
    from benchmarks._noise_generator import NoiseGenerator

    results = {}
    ng = NoiseGenerator(seed=42)

    # ── 10 对隐藏因果对 (无任何因果关键词标记) ──
    # 设计原则:
    # - 每对共享一个隐含的主题域 (如"经济"、"科技"、"政策")
    # - 不使用"导致"、"因为"、"所以"等因果关键词
    # - 包含数值变化 (GaussianDAG 依赖 TF-IDF 数值向量检测)
    hidden_pairs: List[Tuple[str, str, str]] = [
        # (cause, effect, domain)
        ("物价指数同比上涨百分之三点五", "居民消费意愿指数下降百分之八点二", "经济"),
        ("公司宣布大规模裁员两百人", "竞争对手股价上涨百分之五", "经济"),
        ("研发投入大幅增加百分之五十", "产品缺陷率显著下降至百分之零点一", "科技"),
        ("央行基准利率上调五十个基点", "房地产成交量环比下跌百分之十五", "经济"),
        ("国际原油价格突破一百美元每桶", "全球物流运输成本上涨百分之二十", "经济"),
        ("持续极端高温天气超过一个月", "城市用电量创下历史新高", "自然"),
        ("新材料研发取得重大突破", "新能源汽车续航里程翻倍", "科技"),
        ("人口老龄化加速十年百分之三", "医疗消费支出年增速达百分之十二", "社会"),
        ("双减政策全面落地执行", "在线教育行业市值腰斩", "政策"),
        ("AI大模型算力需求指数增长", "芯片厂商营收同比增长百分之八十", "科技"),
    ]

    for n_level in [0, 1, 2, 3]:
        with tempfile.TemporaryDirectory() as tmp:
            engine = SuMemoryLite(storage_path=tmp)

            # ── 阶段 1: 插入真实因果记忆 ──
            all_memories: List[dict] = []
            for i, (cause, effect, domain) in enumerate(hidden_pairs):
                all_memories.append({"id": f"c{i}", "content": cause})
                all_memories.append({"id": f"e{i}", "content": effect})

            # ── 阶段 2: 注入噪声 (交错策略) ──
            if n_level > 0:
                noise_texts = ng.generate(
                    [cause for cause, _, _ in hidden_pairs]
                    + [effect for _, effect, _ in hidden_pairs],
                    noise_level=n_level,
                    noise_mode="semantic" if n_level < 3 else "semantic",
                )
                # 注: noise_level=3 时 generate() 内部自动追加 adversarial
                for k, noise_content in enumerate(noise_texts):
                    all_memories.append({
                        "id": f"n{k}",
                        "content": noise_content,
                    })

            # ── 阶段 3: 统计路径因果发现 ──
            try:
                from su_memory.sdk._causal import CausalEngine
                ce = CausalEngine(min_confidence=0.3)
                stat_pairs = ce.find_causal_pairs(
                    all_memories, use_statistical=True
                )
            except ImportError:
                results[n_level] = 0.0
                continue

            # ── 阶段 4: 评估 ──
            # 检查每对真实因果是否被统计路径检测到
            hits = 0
            for i in range(10):
                cause_content = hidden_pairs[i][0]
                effect_content = hidden_pairs[i][1]

                found_cause = False
                found_effect = False

                for a, b, causal_type, conf in stat_pairs:
                    # 宽松匹配: 原文前 8 字符包含于统计路径结果中
                    cause_id = a.get("id", "")
                    effect_id = b.get("id", "")

                    # 匹配真实记忆 ID
                    if cause_id == f"c{i}" or cause_content[:8] in a.get("content", ""):
                        found_cause = True
                    if effect_id == f"e{i}" or effect_content[:8] in b.get("content", ""):
                        found_effect = True

                if found_cause and found_effect:
                    hits += 1

            results[n_level] = hits / 10.0

    # ── 综合抗噪得分计算 ──
    base = max(results[0], 1e-6)

    # 加权保持率 (越高级噪声权重越大)
    robustness = (
        results[3] / base * 0.50   # 3N 保持率 50%
        + results[1] / base * 0.30  # 1N 保持率 30%
        + results[2] / base * 0.20  # 2N 保持率 20%
    )

    # 分策略抗性
    semantic_resistance = (
        (results[1] + results[2]) / 2.0 / base
    ) if base > 0 else 0
    adversarial_extra_degradation = (
        (results[2] - results[3]) / base
    ) if base > 0 and results[2] > results[3] else 0

    # ── 人类可读解释 ──
    if robustness >= 0.80:
        interpretation = (
            "🟢 检索范式在噪声下极其稳健 (robustness≥0.80)。"
            "v3.6.0 参数化训练优先级可降，检索路径仍有大量余量。"
        )
    elif robustness >= 0.60:
        interpretation = (
            "🟡 检索范式在噪声下表现可接受 (robustness≥0.60)。"
            "3N 噪声下有退化但仍在可用范围。v3.6.0 训练可从容推进。"
        )
    elif robustness >= 0.40:
        interpretation = (
            "🟠 检索范式在噪声下快速退化 (robustness≥0.40)。"
            "建议加速 v3.6.0 参数化转型以弥补检索范式天花板。"
        )
    else:
        interpretation = (
            "🔴 检索范式在噪声下严重退化 (robustness<0.40)。"
            "参数化转型从'优化选项'上升为'生存必需'。"
        )

    return NoiseGradientResult(
        base_score=results[0],
        score_1n=results[1],
        score_2n=results[2],
        score_3n=results[3],
        degradation_1n=results[0] - results[1],
        degradation_2n=results[0] - results[2],
        degradation_3n=results[0] - results[3],
        noise_robustness_score=min(1.0, robustness),
        semantic_noise_resistance=min(1.0, semantic_resistance),
        adversarial_noise_resistance=min(1.0, adversarial_extra_degradation),
        detail=(
            f"0N:{results[0]:.2f}→1N:{results[1]:.2f}"
            f"→2N:{results[2]:.2f}→3N:{results[3]:.2f}"
        ),
        interpretation=interpretation,
    )
```

#### 4.2.5 反脆弱设计 (Anti-Flakiness)

噪声梯度测试涉及随机性——以下措施确保结果可复现:

1. **固定 seed**: `NoiseGenerator(seed=42)` — 每次运行产生完全相同的噪声序列
2. **内容哈希确定性**: 同一原始文本永远映射到相同的噪声变体
3. **宽松的通过阈值**: `noise_robustness_score ≥ 0.60` (而非 0.90)，给统计路径留余地
4. **多次运行均值**: 如果单次运行的噪声梯度结果在边界 (±0.05)，取 3 次运行的中位数

def surface_entities(
    target: str,
    candidate_elements: Optional[List[str]] = None,
) -> List[Tuple[str, RelationType]]:
    """
    从果溯因: 找出哪些元素可能影响 target。
    
    MEMO Step 4 — Entity Surfacing 的能量增强版:
    不仅查找语义关联，还利用五行生克图查找结构关联。
    
    Args:
        target: 目标元素 (效果侧)
        candidate_elements: 候选原因元素列表 (默认五行全部)
        
    Returns:
        [(source_element, relation_type), ...]
        relation_type: ENHANCE (生我) | SUPPRESS (克我) | SAME (同类)
        按关系强度降序
    
    Example:
        >>> surface_entities("fire")
        [("wood", ENHANCE), ("water", SUPPRESS), ("fire", SAME)]
    """
    results = []
    candidates = candidate_elements or list(ENERGY_ENHANCE.keys())
    
    for candidate in candidates:
        if candidate == target:
            results.append((candidate, RelationType.SAME))
            continue
        
        # 生关系: candidate → target (candidate 生 target)
        if ENERGY_ENHANCE.get(candidate) == target:
            results.append((candidate, RelationType.ENHANCE))
        
        # 克关系: candidate → target (candidate 克 target)
        if ENERGY_SUPPRESS.get(candidate) == target:
            results.append((candidate, RelationType.SUPPRESS))
        
        # 被生关系: target → candidate (target 生 candidate，反向)
        if ENERGY_ENHANCE.get(target) == candidate:
            results.append((candidate, RelationType.REVERSE))
    
    # 按关系强度排序
    results.sort(key=lambda x: RELATION_STRENGTH.get(x[1], 0.5), reverse=True)
    return results


def find_reverse_causal_chain(
    effect_element: str,
    depth: int = 2,
) -> List[List[Tuple[str, RelationType]]]:
    """
    从果溯因的因果链搜索 (深度优先)。
    
    depth=1: 直接影响 effect 的元素
    depth=2: 间接影响 (通过 1 个中间元素)
    depth=3: 深层影响 (通过 2 个中间元素)
    
    Returns:
        因果链列表，每条链是 [(element, relation), ...]
    """
    chains = []
    
    # depth 1: 直接表面
    direct = surface_entities(effect_element)
    for elem, rel in direct:
        chains.append([(elem, rel)])
    
    if depth >= 2:
        for elem, _ in direct:
            sub = surface_entities(elem)
            for sub_elem, sub_rel in sub:
                if sub_elem != effect_element:  # 避免循环
                    chains.append([(sub_elem, sub_rel), (elem, RelationType.ENHANCE)])
    
    # 去重
    seen = set()
    unique_chains = []
    for chain in chains:
        key = "→".join(c[0] for c in chain)
        if key not in seen:
            seen.add(key)
            unique_chains.append(chain)
    
    return unique_chains
```

**背景**: LeJEPA (2511.08544v2) 证明了 **各向同性高斯是 JEPA 嵌入的最优分布**，SIGReg (Sketched Isotropic Gaussian Regularization) 仅 ~50 行代码即可实现。su-memory 将其适配为嵌入后处理层。

```python
"""
su-memory v3.5.0 — SIGReg Embedding Regularizer
================================================
基于 LeJEPA (2511.08544v2) 的 Sketched Isotropic Gaussian Regularization。

核心洞察: 嵌入向量的各向同性程度与下游检索质量正相关。
SIGReg 通过最小化与各向同性高斯的矩差来正则化嵌入。

数学:
  L_SIGReg(z) = ||mean(z)||² + λ · ||cov(z) - I||²
  其中 z 为归一化后的嵌入向量。

复杂度: O(d²) per batch, d=768 时 < 1ms
行数: ~50 行 (对齐 LeJEPA 设计理念)
"""

import numpy as np


class SIGReg:
    """
    Sketched Isotropic Gaussian Regularizer.
    
    用途:
    1. 嵌入后处理: FAISS 索引前对嵌入做各向同性正则
    2. 训练时正则: 记忆模型训练时作为辅助损失
    
    超参数:
    - lambda_reg: 正则强度 (默认 0.01)
    - use_sketch: 是否使用 sketched 加速 (默认 True, SVD 近似)
    """
    
    def __init__(
        self,
        lambda_reg: float = 0.01,
        use_sketch: bool = True,
        sketch_dim: int = 64,
    ):
        self.lambda_reg = lambda_reg
        self.use_sketch = use_sketch
        self.sketch_dim = sketch_dim
    
    def __call__(self, embeddings: np.ndarray) -> np.ndarray:
        """
        对嵌入矩阵做各向同性正则化。
        
        Args:
            embeddings: shape=(n, d) 嵌入矩阵
            
        Returns:
            正则化后的嵌入矩阵, shape=(n, d)
        """
        return self.regularize(embeddings)
    
    def regularize(self, embeddings: np.ndarray) -> np.ndarray:
        """
        两步正则:
        1. 零均值化 (mean shift)
        2. 协方差白化 (cov → I)
        """
        z = embeddings.copy().astype(np.float64)
        n, d = z.shape
        
        # ① 零均值化
        z_mean = z.mean(axis=0, keepdims=True)
        z = z - z_mean
        
        # ② 白化 (协方差 → 单位矩阵)
        if self.use_sketch and d > self.sketch_dim:
            # Sketched: 仅在低维子空间做白化
            sketch = np.random.randn(d, self.sketch_dim) / np.sqrt(self.sketch_dim)
            z_sketch = z @ sketch  # (n, sketch_dim)
            cov_sketch = z_sketch.T @ z_sketch / (n - 1)
            
            # 特征分解 + 白化
            eigvals, eigvecs = np.linalg.eigh(cov_sketch)
            eigvals = np.maximum(eigvals, 1e-6)
            whitening = eigvecs @ np.diag(1.0 / np.sqrt(eigvals)) @ eigvecs.T
            
            z_whitened = z_sketch @ whitening
            # 投影回原空间
            z_reg = z_whitened @ sketch.T
        else:
            # 完整白化
            cov = z.T @ z / (n - 1)
            eigvals, eigvecs = np.linalg.eigh(cov)
            eigvals = np.maximum(eigvals, 1e-6)
            whitening = eigvecs @ np.diag(1.0 / np.sqrt(eigvals)) @ eigvecs.T
            z_reg = z @ whitening
        
        # ③ 正则化强度插值 (保留部分原始方向)
        result = z * (1.0 - self.lambda_reg) + z_reg * self.lambda_reg
        
        # L2 归一化
        norms = np.linalg.norm(result, axis=1, keepdims=True)
        norms = np.maximum(norms, 1e-10)
        result = result / norms
        
        return result.astype(embeddings.dtype)
    
    def compute_isotropy_score(self, embeddings: np.ndarray) -> float:
        """
        计算嵌入的各向同性得分。
        
        0 = 完全退化 (所有向量相同方向)
        1 = 完全各向同性 (均匀分布在高维球面上)
        
        使用协方差矩阵的条件数:
        score = 1 / cond(cov)
        """
        z = embeddings.copy().astype(np.float64)
        z_mean = z.mean(axis=0, keepdims=True)
        z = z - z_mean
        cov = z.T @ z / (z.shape[0] - 1)
        
        eigvals = np.linalg.eigvalsh(cov)
        eigvals = np.maximum(eigvals, 1e-10)
        
        cond = eigvals.max() / eigvals.min()
        return min(1.0, 1.0 / max(cond, 1e-10))
```
